function Toggle(tabToShow, divAreaName){
    var numerOfTabs = 0;
    for (var loop = 0; loop < document.getElementsByTagName('div').length; loop++) {
        var DivIdName = "document.getElementsByTagName('DIV')[" + loop + "].id";
        if (eval(DivIdName) && eval(DivIdName).indexOf(divAreaName) != -1) {
            numerOfTabs++;
        }
    }
    for (var loop = 1; loop <= numerOfTabs; loop++) {
        var currentTabId = divAreaName + loop;
        var action;
        if (loop == tabToShow) {
            action = "block";
        } else {
            action = "none";
        }
        if (browserType == "NN") {
            eval("document." + currentTabId + ".visibility='" + action + "'");
        } else if (browserType == "IE") {
            eval("document.all." + currentTabId + ".style.display='" + action + "'");
        } else if (browserType == "MO" || browserType == "OP") {
            eval("document.getElementById('" + currentTabId + "').style.display='" + action + "'");
        }
    }
}

	function validateDates(fromDate,toDate,fromdateMsg,endDateMsg){
		
		var objFromDate = fromDate;
		var objToDate = toDate;
		
		if((objFromDate=='' || objFromDate==null) && (objToDate!='' && objToDate!=null))
		{
			
			$("#errormsg").empty();
			$("#errormsg").append("Please select a value for the"+" "+fromdateMsg);
			return false; 
		}
		if((objToDate=='' || objToDate==null) && (objFromDate!='' && objFromDate!=null))
		{
			$("#errormsg").empty();
			$("#errormsg").append("Please select a value for the"+" "+endDateMsg);
			return false; 
		}
		
		var fromdate = objFromDate.split("-");
		var todate = objToDate.split("-");
		var date1 = new Date(fromdate[0],(fromdate[1]-1),fromdate[2]);
		var date2 = new Date(todate[0],(todate[1]-1),todate[2]);
		var date3 = new Date();
		
		if(date1 > date2)
		{
	
		$("#errormsg").empty();
		$("#errormsg").append("The"+" "+fromdateMsg+" "+"should be earlier than the"+" "+endDateMsg);
		return false; 
		}
		else if(date1 > date3)
		{
		
		$("#errormsg").empty();
		$("#errormsg").append("The"+" "+fromdateMsg+" "+"should be earlier than or equal to the current date.");
		return false; 
		}
		else if(date2 > date3) 
		{
		
		$("#errormsg").empty();
		$("#errormsg").append("The"+" "+endDateMsg+" "+"should be earlier than or equal to the current date.");
		return false; 
		}
		return true;

		}
